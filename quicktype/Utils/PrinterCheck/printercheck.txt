1 REM ** CHECK FOR PRINTER **
2 REM *    ON=75 : OFF=71   *
3 REM *    GARY FURR        *
4 REM ***********************
30 CLS
40 IFPEEK(3)=75THEN70
50 PRINT"PRINTER OFF/LINE, TURN IT ON"
55 GOSUB100:CLS:GOSUB100
60 GOTO30
70 PRINT"PRINTER ON-LINE"
75 GOSUB100:CLS:GOSUB100
80 GOTO30
100 FORT=1TO1000:NEXTT:RETURN
