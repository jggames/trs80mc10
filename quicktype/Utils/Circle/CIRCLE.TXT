5 CLS:PRINT"HIT A KEY. THEN ENTER X,Y,R,C"
10 I$=INKEY$:ON1-(I$<>"")GOTO10:CLS
20 INPUTCX,CY,R,C:CLS0:GOSUB30:GOTO10
30 XC=0:YC=R:PC=3-2*R 
35 GOSUB70
40 ON1-(PC<0)GOTO45:PC=PC+4*XC+6:GOTO50
45 PC=PC+4*(XC-YC)+10:YC=YC-1   
50 XC=XC+1:ON1-(XC<YC)GOTO60,35
60 ON1-(XC<>YC)GOSUB70:RETURN
70 XX=CX+XC:YY=CY+YC:SET(XX,YY,C):XX=CX-XC:YY=CY+YC:SET(XX,YY,C):XX=CX+XC:YY=CY-YC:SET(XX,YY,C):XX=CX-XC:YY=CY-YC:SET(XX,YY,C)
75 XX=CX+YC:YY=CY+XC:SET(XX,YY,C):XX=CX-YC:YY=CY+XC:SET(XX,YY,C):XX=CX+YC:YY=CY-XC:SET(XX,YY,C):XX=CX-YC:YY=CY-XC:SET(XX,YY,C)
79 RETURN
