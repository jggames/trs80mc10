10 DIM X,Y,J,A,C,P,I,N,Q
20 N=31:Q=15
30 CLS 0
40 FOR J=1 TO 8 STEP .35
50 FOR A=1 TO 8 STEP .55
60 P=A*J
70 X=P*COS(J)+N
80 Y=P*SIN(J)+Q
90 IF X<0 THEN X=0
100 IF Y<0 THEN Y=0
110 IF X>64 THEN X=63
120 IF Y>31 THEN Y=31
130 SET(X,Y,C)
140 NEXT
150 C=C+1:IF C>8 THEN C=1
160 NEXT
170 N=RND(63):Q=RND(31)
180 GOTO 30
