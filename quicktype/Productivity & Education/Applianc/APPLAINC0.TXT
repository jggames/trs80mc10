5 : REJVI "COST OF APPLIANCE"
7 : REM "KILOWATT-HOURS"
10 :"A":INPUT"PREVIOUS METER READING?"; A 
20 :INPUT"PRESENT METER READING?";B:IFB <ALET 8=8+10000
30 :C=B-A:PRINT"CONSUMPTION = ";C
40 :IFC<550 THEN 90
50 :D=550:E=D*.04589
60 :F=C-D:G=F*a03089
70 :I=5.00+E+G
80 :PRINT"COST OF POWER =$";I:GOT0100
90 :I=5.00+(C*.04589):GOT080
100 :"S":INPUT"WATTAGE OF APPLIANCE ?"; J:IFJ=OGOSUB400:GOSUB300 120 :K=(J/1000)*M:K=INTCK*10000)/10000
130 :PRINT"COST PER HOUR =$";K
140 :PRINT"FOR CONT. USE,<SHFT D>"
150 :GOT0100
200 :P=S*365:Q=P/12:P=INT(P*10000)/1000 O:Q=INT(Q*10000)/10000 210 :PRINT"COST PER DAY =$";S
220 :PRINT"COST PER MONTH =$";Q
230 :PRINT"COST PER YEAR =$";P
240 :PRINT"TO RUN AGAIN <SHFT S>"
250 :END
300 :M=(I-5.00)/C:RETURN
400 :INPUT"AMPERAGE OF APPLIANCE ?";L
410 :J=L*120:RETURN
500 :"D":INPUT"HOURS ON PER DAY? ";R
510 :S=K*R:GOTO 200
