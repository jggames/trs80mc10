10 REM **** CUBIST ART ****
20 REM
30 REM By D. R. Hunt        December 1979
40 REM
50 REM For Nascom 1 or 2, using Nas-sys 1 and
60 REM     NAS-GRA V.3 Graphics.
70 REM
80 CLS:T$="CUBIST ART":FORI=1TOLEN(T$)
90 POKE3035+I,ASC(MID$(T$,I,1)):NEXT
100 A=RND(1)*95:B=RND(1)*43+1:C=RND(1)*95
110 D=RND(1)*43+1:A1=-1:A2=-1:IFA<C THENA1=1
120 IFB<D THENA2=1
130 FORI=ATOCSTEP1A:FORJ=BTODSTEPA2
140 A3=POINT(I,J):IFA3=1THENRESET(I,J):GOTO160
150 SET(I,J)
160 NEXT:NEXT:GOTO100
