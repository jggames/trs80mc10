10 REM FILENAME: "C5P9"
20 REM FUNCTION: DRAW A VERY SIMPLE CHRISTMAS TREE
30 REM AUTHOR: JOHN P. GRILLO "TECHNIQUES OF BASIC" 1981 
40 CLS:POKE49151,64:CLEAR 200:MC=16384:PRINT"WANT A TREE, EH! HOW 'BOUT THIS ONE?"
50 M=32:R1=0:R2=0:Z=RND(-(PEEK(9)*256+PEEK(10)))
60 FOR I=1 TO 12
70 IF I>1 THEN M=M-RND(3):R1=RND(5)-3:R2=RND(5)-3
80 PRINTTAB((M-R1)/2)
90 L=65-M*2+R2:M$="":FORZ=1TO32:IFRND(9)<8THENM$=M$+CHR$(127+RND(16)):GOTO92
91 M$=M$+CHR$(128+RND(127))
92 NEXT
100 PRINT LEFT$(M$,L/2)
110 NEXT I
120 PRINT TAB(29/2)"merry"
130 PRINT TAB(29/2)"xmas!";:Z=(PEEK(17024)AND1)*256+PEEK(17025)-1+MC:POKEZ,PEEK(Z)-64
140 FORZ=1TO2500
150 IFINKEY$<>""THENZ=2500
160 NEXT
170 PRINT:PRINT"BIT OF A CHARLIE BROWN TREE, EH!";
180 FORZ=1TO2500:NEXT
190 CLS:PRINT"HOW ABOUT THIS ONE?"
200 GOTO50
